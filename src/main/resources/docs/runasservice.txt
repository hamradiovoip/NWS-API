[Unit]
Description=WeatherAPI
After=syslog.target network.target

[Service]
User=user
Group=mygroup

Environment="JAVA_HOME=/usr/bin/java"

ExecStart=/usr/bin/java -jar /usr/weatherapi/WeatherAPI-0.0.1-SNAPSHOT.jar
ExecStop=/bin/kill -15 $MAINPID
SuccessExitStatus=143

[Install]
WantedBy=multi-user.target

service weatherapi start

chown user:user /var/weatherapi/WeatherAPI-0.0.1-SNAPSHOT.jar

to restart at boot:
 update-rc.d weatherapi defaults
